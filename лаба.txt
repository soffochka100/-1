#include <stdio.h>
#include <math.h>

float z(float x, float a)
{
float w=sqrt((a-x)/x);
float g=cos((a*a)/x);
float q=cos((x*cbrt(a))/(a+2.5));

if (w>g && w>q) return w;
else if (g>w && g>q) return g;
else return q;

}

int main()
{
for(float x=1; x<=3; x+=1.3)
{
for(float a=2;a<=4;a+=1.3)
{
printf("x=%f a=%f z(x,a)=%f\n",x,a,z(x,a));
}
}
return 0;
}
